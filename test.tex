\documentclass{beamer}

\usepackage{namedoverlays}

\begin{document}

\overlays{
  first={
    first sub,
    other sub,
  },
  other={
    first sub,
    other sub,
  },
}

\begin{frame}{Some slide}
  These items demonstrate several possible overlay specifications and when they
  are active.

  \begin{itemize}
  \item<\olto{first}{other/first sub}>
    \texttt{first} $\to$ \texttt{other/first sub}
  \item<\ol{first}> \texttt{first}
    \begin{itemize}
    \item<\ol{first/first sub}> \texttt{first sub}
    \item<\ol{first/other sub}> \texttt{other sub}
    \end{itemize}
  \item<\ol{other}> \texttt{other}
    \begin{itemize}
    \item<\ol{other/first sub}> \texttt{first sub}
    \item<\ol{other/other sub}> \texttt{other sub}
    \end{itemize}
  \item<\olto{first/other sub}{}>
    \texttt{first/other sub} $\to$ last slide
  \item<\ol{first/first sub},\ol{other/other sub}>
    \texttt{first/first sub} $\lor$ \texttt{other/other sub}
  \end{itemize}
\end{frame}

\end{document}
